        function addUser() {

            const newName = prompt("請輸入新使用者的姓名:");

            if (!newName) return;

            if (users.some(user => user.name === newName)) {

                alert("錯誤：此使用者名稱已存在！");

                return;

            }

            const newRole = prompt(`請輸入 ${newName} 的角色 (農夫/技師/管理者):`, "農夫");

            if (!newRole) return;

            const newEmail = prompt(`請輸入 ${newName} 的電子郵件:`);

            if (!newEmail) return;

 

            users.push({ name: newName, role: newRole, email: newEmail });

            renderUsersTable();

            alert(`使用者 ${newName} (${newRole}) 已新增。`);

        }

 

        function editUser(name) {

            const userIndex = users.findIndex(user => user.name === name);

            if (userIndex === -1) return;

            const user = users[userIndex];

 

            const newRole = prompt(`編輯 ${name} 的角色 (目前: ${user.role})，請輸入新角色:`, user.role);

            if (!newRole) return;

            const newEmail = prompt(`編輯 ${name} 的電子郵件 (目前: ${user.email})，請輸入新電子郵件:`, user.email);

            if (!newEmail) return;

 

            users[userIndex] = { ...user, role: newRole, email: newEmail };

            renderUsersTable();

            alert(`使用者 ${name} 的資料已更新。`);

        }
